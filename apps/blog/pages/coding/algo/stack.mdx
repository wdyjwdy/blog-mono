## 有效括号
```js
function isValid(s: string): boolean {
    let set = new Set(["()", "[]", "{}"])
    let stack = []
    for (let c of s) {
        if (set.has(stack.at(-1) + c)) {
            stack.pop()
        } else {
            stack.push(c)
        }
    }
    return stack.length === 0
};
```
## 每日温度
```js
function dailyTemperatures(temperatures: number[]): number[] {
    let len = temperatures.length
    let res = Array(len).fill(0)
    let stack = []
    for (let i = 0; i < len; i++) {
        while (stack.length > 0 && temperatures[stack.at(-1)] < temperatures[i]) {
            let index = stack.pop()
            res[index] = i - index
        }
        stack.push(i)
    }
    return res
};
```

```js
function nextGreaterElement(nums1: number[], nums2: number[]): number[] {
    let map = new Map()
    let stack = []
    for (let [i, v] of nums2.entries()) {
        while (stack.length > 0 && nums2[stack.at(-1)] < v) {
            map.set(nums2[stack.pop()], v)
        }
        stack.push(i)
    }
    return nums1.map(x => map.get(x) ?? -1)
};
```

## 基本计算器 II
```js
function calculate(s: string): number {
    let tokens = s
        .replace(/ /g, '') // 去掉空格
        .split(/([^\d])/g) // 分离数字
        .map(x => /\d/.test(x) ? Number(x) : x) // 转为整数
    let nums = []
    let op = '+'
    for (let t of tokens) {
        if (typeof t === 'number') {
            if (op === '+') nums.push(t)
            else if (op === '-') nums.push(-t)
            else if (op === '*') nums.push(nums.pop() * t)
            else if (op === '/') nums.push(~~(nums.pop() / t))
        } else {
            op = t
        }
    }
    return nums.reduce((a, b) => a + b)
};
```
## 逆波兰表达式求值
```js
function evalRPN(tokens: string[]): number {
    let nums = []
    for (let t of tokens) {
        if (/\d/.test(t)) {
            nums.push(Number(t))
        } else {
            let r = nums.pop()
            let l = nums.pop()
            if (t === '+') nums.push(l + r)
            else if (t === '-') nums.push(l - r)
            else if (t === '*') nums.push(l * r)
            else if (t === '/') nums.push(~~(l / r))
        }
    }
    return nums[0]
};
```
